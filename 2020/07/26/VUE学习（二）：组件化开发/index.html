2.3、子传父（自定义事件$emit）

<div id="app">
  <!--不写参数默认传item过去-->
  <cpn @item-click="cpnClick"></cpn> //使用方法名绑定自己的方法
</div>

<template id="cpn">
  <!--有多个参数需要显示，必须用东西包起来-->
  <div>
    <!--通过自定义事件，把参数传下去-->
    <button v-for="item in categorys" @click="btnClick(item)">
      {{item.name}}
    </button>
  </div>
</template>

<script src="../js/vue.js"></script>
<script>
  const cpn = {
    template: '#cpn',
    data() {
      return {
        categorys: [
          {id: 'aaa',name: '热门推荐'},
          {id: 'bbb',name: '手机数码'},
          {id: 'ccc',name: '家用家电'},
          {id: 'ddd',name: '电脑办公'}
        ]
      }
    },
    methods: {
      //在这里把具体参数传给父组件
      btnClick(item) {
        //(方法名，参数)
        this.$emit('item-click',item);
      }
    }
  }

  const app = new Vue({
    el: "#app",
    data: {
      message: '你好啊'
    },
    components: {
      cpn
    },
    methods: {
      //取得具体参数
      cpnClick(item){
        console.log(item.name);
      }
    }
  })
</script>

2.4、父子组件通信案例

<div id="app">
  <cpn :number1="num1" :number2="num2" @dnumber1tofather="num1change" @dnumber2tofather="num2change"></cpn>
</div>

<template id="cpn">
  <div>
    <h2>父组件-传给props：{{number1}}</h2>
<!--    <input type="text" v-model="dnumber1">-->
    <h2>子组件-props赋值给data:{{dnumber1}}</h2>
    <input type="text" :value="dnumber1" @input="num1Input">  <!--拆分,通过方法发送$emit()-->

    <hr>

    <h2>父组件-传给props：{{number2}}</h2>
<!--    <input type="text" v-model="dnumber2">-->
    <h2>子组件-props赋值给data:{{dnumber2}}</h2>
    <input type="text" :value="dnumber2" @input="num2Input">  <!--拆分,通过方法发送$emit()-->
  </div>
</template>

<script src="../js/vue.js"></script>
<script>
  const app = new Vue({
    el: "#app",
    data: {
      num1: 1,
      num2: 0
    },
    methods: {
      num1change(value) {
        this.num1 = parseFloat(value);
      },
      num2change(value) {
        this.num2 = parseFloat(value);
      }
    },
    components: {
      cpn: {
        template: '#cpn',
        //不能直接绑定到props里，会报错，需要绑定到data里
        props: {
          number1: Number,
          number2: Number
        },
        data() {
          return {
            dnumber1: this.number1,
            dnumber2: this.number2
          }
        },
        methods: {
          num1Input(event) {
            this.dnumber1 = event.target.value;
            this.$emit('dnumber1tofather',this.dnumber1);

            this.dnumber2 = this.dnumber1*100;  //num1 * 100 -> num2  ，控制num2
            this.$emit('dnumber2tofather',this.dnumber2);
          },
          num2Input(event) {
            this.dnumber2 = event.target.value;
            this.$emit('dnumber2tofather',this.dnumber2);

            this.dnumber1 = this.dnumber2/100;  //num1 * 100 -> num2  ，控制num2
            this.$emit('dnumber1tofather',this.dnumber1);
          }
        }
      }
    }
  })
</script>

需求：
1、父传子
2、子双向绑定
3、子同步父
4、通过子参数1修改子参数2，子参数参数1是子参数2的100倍，子参数2是子参数1的1/100，同时子同步父

总结：
1、父传子，子取得的参数不能直接绑定到子的props中，会报错，需要绑定到子的data中。
2、拆分v-model为v-bind和v-on，并通过@input="num1Input"实时监听输入的数据，绑定方法，通过$emit修改父
3、父接受，修改父参数。
4、父参数修改时要注意类型不同的问题，需要把string转化为int

2.5、父子组件通信案例watch的使用

//不拆分，直接双向绑定
<input type="text" v-model="dnumber1">
<input type="text" v-model="dnumber2">

//使用watch自己监听变量是否改变：有两个参数（newVaule，oldValue）
watch: {
    //配合v-model，实时绑定，监听变量是否发生改变
    dnumber1(newValue) {
        this.dnumber2 = this.dnumber1*100;        //一旦改变，就可以监听到dnumber2
        this.$emit('dnumber1tofather',newValue);  //改变自身的props
    },
    dnumber2(newValue) {
        this.dnumber1 = this.dnumber2/100;       //num1 * 100 -> num2  ，控制num2
        this.$emit('dnumber2tofather',newValue);
    }
}
